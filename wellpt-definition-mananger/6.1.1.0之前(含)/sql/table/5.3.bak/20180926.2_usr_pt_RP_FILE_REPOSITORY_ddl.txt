-- Create table
create table RP_FILE_REPOSITORY
(
  uuid        VARCHAR2(64) primary key ,
  create_time TIMESTAMP(6),
  creator     VARCHAR2(64),
  modifier    VARCHAR2(64),
  file_id     VARCHAR2(64),
  file_name   VARCHAR2(120 CHAR) not null,
  modify_time TIMESTAMP(6),
  rec_ver     NUMBER(10),
  content     CLOB
)
tablespace OA_DATA
  pctfree 10
  initrans 1
  maxtrans 255
  storage
  (
    initial 64K
    next 1M
    minextents 1
    maxextents unlimited
  );
